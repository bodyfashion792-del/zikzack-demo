<script>
let usdt=10000;
let btc=1;
let price=43250;
let logged=false;

function show(id){
  document.querySelectorAll('.card').forEach(c=>c.classList.add('hidden'));
  document.getElementById(id).classList.remove('hidden');
}

function login(){
  logged=true;
  alert("Demo Login Successful");
  show('trade');
}

function update(){
  document.getElementById("usdt").innerText=usdt.toFixed(2);
  document.getElementById("btc").innerText=btc.toFixed(4);
}

function buy(){
  if(!logged){alert("Please login first");return;}
  let cost=price*0.01;
  if(usdt>=cost){
    usdt-=cost;
    btc+=0.01;
    update();
    alert("Demo Buy Successful");
  }else alert("Not enough USDT");
}

function sell(){
  if(!logged){alert("Please login first");return;}
  if(btc>=0.01){
    btc-=0.01;
    usdt+=price*0.01;
    update();
    alert("Demo Sell Successful");
  }else alert("Not enough BTC");
}
